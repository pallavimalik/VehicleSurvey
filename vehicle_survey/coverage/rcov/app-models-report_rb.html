110   def self.group_by_time_period(vehicle_list, time_period_duration)
 
111     vehicle_by_time = []
 
112     time_period = nil
 
113     cur_time_period = 0
 
114     vehicle_list[0].each_cons(2) do |prev_vehicle, next_vehicle|
 
115       if next_vehicle.first < prev_vehicle.first
 
116         vehicle_by_time << time_period if time_period != nil
 
117         cur_time_period = 0
 
118         time_period = [[]]
 
119       elsif next_vehicle.first > (cur_time_period + 1) * time_period_duration
 
120         time_period = [[]] if time_period.nil?
 
121         time_period[cur_time_period] << prev_vehicle if prev_vehicle != time_period[cur_time_period][0]
 
122         while next_vehicle.first > (cur_time_period + 1) * time_period_duration
 
123           cur_time_period += 1
 
124           time_period[cur_time_period] = []
 
125         end
 
126         time_period[cur_time_period] << next_vehicle if next_vehicle != time_period[cur_time_period][0]
 
127       else
 
128         time_period = [[]] if time_period.nil?
 
129         time_period[cur_time_period] << prev_vehicle if prev_vehicle != time_period[cur_time_period][0]
 
130       end
 
131     end
 
132     time_period[cur_time_period] << vehicle_list[0].last
 
133     vehicle_by_time << time_period
 
134   end
